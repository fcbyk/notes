html
<nav>
    <!-- 两列表嵌套 -->
    <ul class="menus">
        <li class="li">
            <!-- 图标库,名称 -->
            <i class="fa fa-comment"></i> 名称

            <!-- 子列表隐藏->宽度设为0 -->
            <ul><li>内容</li></ul>
        </li>
    </ul>
</nav>
scss
nav {
	//形状与定位方式
    .menus {
        // 字体设置
        .li {
            &:hover {
				// 鼠标悬浮时，列表项背景颜色改变，
                ul {
                    // 子列表宽度变化
                }
            }
            .fa { // 图标设置 }
            ul {
                // 子列表形状与定位方式，过渡效果
                li {
                    // 列表项设置
                    &:hover {
                        // 背景颜色变化
                    }
                }
            }
        }
    }
}
